DROP TABLE IF EXISTS wifitrg;
DROP TABLE IF EXISTS wifibc;

CREATE TABLE wifibc
(
   Id INT(20) NOT NULL AUTO_INCREMENT,
   Scaner VARCHAR(30),
   SSID VARCHAR(33) NOT NULL,
   Networkmac VARCHAR(23) NOT NULL,
   TIME TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP,
	SIG INT(20) NOT  NULL,
   LAT FLOAT(13, 10) NOT NULL,
   LON FLOAT(13, 10) NOT NULL,
   Securite VARCHAR(50),
   PRIMARY KEY (Id)
);

CREATE TABLE wifitrg
(
   Id INT(20) NOT NULL AUTO_INCREMENT,
   DernierScaner VARCHAR(30),
   PremierScaner VARCHAR(30),
   SSID VARCHAR(33) NOT NULL,
   Networkmac VARCHAR(23) NOT NULL,
	SIGMOY FLOAT,
   Premiereaparition TIMESTAMP,
   derniereaparition TIMESTAMP,
   LAT FLOAT( 13, 10) NOT NULL,
   LON FLOAT( 13, 10) NOT NULL,
   Securite VARCHAR(50),
   NBBC INT,
   PRIMARY KEY (Id)
);
